

### 2. edit urls

``` python
# urls.py

# ...

from contact.views import ContactWizard, FORMS

urlpatterns = [
    url(r'^contact/', ContactWizard.as_view(FORMS)),
    url(r'^done/', TemplateView.as_view(template_name='done.html')),
]
```
